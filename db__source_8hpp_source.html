30using BindingVariant = std::variant<int, double, std::string>;
31class DBSource {
32private:
33 std::unique_ptr<SQLite::Database> db = nullptr;
34 std::string path = "";
35
36public:
37 DBSource() {}
38 ~DBSource() = default;

44 void ConnectToDatabase(const std::string &p) {
45 path = p;
46 db = std::make_unique<SQLite::Database>(p, SQLite::OPEN_READWRITE |
47 SQLite::OPEN_CREATE);
48 }
49
50 std::string GetName() const {
51 std::filesystem::path p = path;
52 return p.stem();
53 }
54
55 void
56 Select(const std::string &query,
57 std::function<void(std::any &storage, const SQLite::Statement &stmt)>
58 callback,
59 std::any &storage,
60 const std::unordered_map<int, BindingVariant> &bindings = {}) {
61 try {
62 SQLite::Statement stmt(*db, query);
63
64 for (const auto &[index, value] : bindings) {
65 if (value.index() == 0) {
66 stmt.bind(index, std::get<int>(value));
67 } else if (value.index() == 1) {
68 stmt.bind(index, std::get<double>(value));
69 } else {
70 stmt.bind(index, std::get<std::string>(value));
71 }
72 }
73
74 SQLite::Transaction transaction(*db);
75
76 while (stmt.executeStep()) {
77 callback(storage, stmt);
78 }
79
80 transaction.commit();
81 } catch (const std::exception &e) {
82 throw std::runtime_error("Error executing query: " + query + "\n" +
83 e.what());
84 }
85 }
86
87 void BatchExecute(const std::string &query,
88 const std::vector<std::unordered_map<int, BindingVariant>>
89 &bindings_batch = {}) {
90 try {
91 SQLite::Transaction transaction(*db);
92 SQLite::Statement stmt(*db, query);
93
94 for (auto &bindings : bindings_batch) {
95 for (const auto &[index, value] : bindings) {
96 if (value.index() == 0) {
97 stmt.bind(index, std::get<int>(value));
98 } else if (value.index() == 1) {
99 stmt.bind(index, std::get<double>(value));
100 } else {
101 stmt.bind(index, std::get<std::string>(value));
102 }
103 }
104 stmt.exec();
105 stmt.reset();
106 }
107 transaction.commit();
108
109 } catch (const std::exception &e) {
110 throw std::runtime_error("Error executing query: " + query + "\n" +
111 e.what());
112 }
113 }
114};
